%% Two Parameters mu and sigma
\pgfmathdeclarefunction{gauss}{2}{
  \pgfmathparse{1/(#2*sqrt(2*pi))*exp(-((x-#1)^2)/(2*#2^2))}
}
\pgfmathdeclarefunction{logistic}{1}{
  \pgfmathparse{1/(1+ exp(-#1*x))}
}

\begin{figure}[ht]
	\centering  
		\label{fig:gaussian_distribution}
	\begin{tikzpicture}
		\begin{axis}[
		  every axis plot post/.append style=
		  {
		  		mark=none,domain=-2:2,samples=200,smooth
		  }, % All plots: from -2:2, 500 samples, smooth, no marks
		  axis x line*=bottom, % no box around the plot, only x and y axis
		  axis y line*=left, % the * suppresses the arrow tips
		  enlargelimits=upper,
		  legend columns=-1,
		  legend style = {at={(0.5, 0.93)}, anchor=south, draw=none}
		] % extend the axes a bit to the right and top
		  \addplot [blue, very thick] {gauss(0,0.25)};
		  \addplot [red, very thick] {logistic(3)};
		  \legend{Gaussian, Logistic}
		\end{axis}
	\end{tikzpicture}
	\caption{Gaussian Distribution and Logistic Function}
\end{figure}
